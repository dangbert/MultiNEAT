55 inline double af_sigmoid_unsigned(double aX, double aSlope, double aShift)
56 {
57  return 1.0 / (1.0 + exp( - aSlope * aX - aShift));

60 inline double af_sigmoid_signed(double aX, double aSlope, double aShift)
61 {
62  double tY = af_sigmoid_unsigned(aX, aSlope, aShift);

297 void NeuralNetwork::ActivateFast()
298 {
299  // Loop connections. Calculate each connection's output signal.
300  for (unsigned int i = 0; i < m_connections.size(); i++)
301  {
302  m_connections[i].m_signal =
303  m_neurons[m_connections[i].m_source_neuron_idx].m_activation
304  * m_connections[i].m_weight;
305  }
306  // Loop the connections again. This time add the signals to the target neurons.
307  // This will largely require out of order memory writes. This is the one loop where
308  // this will happen.
309  for (unsigned int i = 0; i < m_connections.size(); i++)
310  {
311  m_neurons[m_connections[i].m_target_neuron_idx].m_activesum +=
312  m_connections[i].m_signal;
313  }
314  // Now loop nodes_activesums, pass the signals through the activation function
315  // and store the result back to nodes_activations
316  // also skip inputs since they do not get an activation
317  for (unsigned int i = m_num_inputs; i < m_neurons.size(); i++)
318  {
319  double x = m_neurons[i].m_activesum;
320  m_neurons[i].m_activesum = 0;
321  // Apply the activation function
322  double y = 0.0;
323  y = af_sigmoid_unsigned(x, m_neurons[i].m_a, m_neurons[i].m_b);
324  m_neurons[i].m_activation = y;
325  }

668 void NeuralNetwork::Adapt(Parameters& a_Parameters)
669 {
670  // find max absolute magnitude of the weight
671  double t_max_weight = -999999999;
672  for (unsigned int i = 0; i < m_connections.size(); i++)
673  {
674  if (fabs(m_connections[i].m_weight) > t_max_weight)
675  {
676  t_max_weight = fabs(m_connections[i].m_weight);
677  }
678  }
679 
680  for (unsigned int i = 0; i < m_connections.size(); i++)
681  {
683  // modify weight of that connection
685  double t_incoming_neuron_activation =
686  m_neurons[m_connections[i].m_source_neuron_idx].m_activation;
687  double t_outgoing_neuron_activation =
688  m_neurons[m_connections[i].m_target_neuron_idx].m_activation;
689  if (m_connections[i].m_weight > 0) // positive weight
690  {
691  double t_delta = (m_connections[i].m_hebb_rate
692  * (t_max_weight - m_connections[i].m_weight)
693  * t_incoming_neuron_activation
694  * t_outgoing_neuron_activation)
695  + m_connections[i].m_hebb_pre_rate * t_max_weight
696  * t_incoming_neuron_activation
697  * (t_outgoing_neuron_activation - 1.0);
698  m_connections[i].m_weight = (m_connections[i].m_weight + t_delta);
699  }
700  else if (m_connections[i].m_weight < 0) // negative weight
701  {
702  // In the inhibatory case, we strengthen the synapse when output is low and
703  // input is high
704  double t_delta = m_connections[i].m_hebb_pre_rate
705  * (t_max_weight - m_connections[i].m_weight)
706  * t_incoming_neuron_activation
707  * (1.0 - t_outgoing_neuron_activation)
708  - m_connections[i].m_hebb_rate * t_max_weight
709  * t_incoming_neuron_activation
710  * t_outgoing_neuron_activation;
711  m_connections[i].m_weight = -(m_connections[i].m_weight + t_delta);
712  }
713 
714  Clamp(m_connections[i].m_weight, -a_Parameters.MaxWeight,
715  a_Parameters.MaxWeight);
716  }

733 void NeuralNetwork::RTRL_update_gradients()
734 {
735  // for every neuron
736  for (unsigned int k = m_num_inputs; k < m_neurons.size(); k++)
737  {
738  // for all possible connections
739  for (unsigned int i = m_num_inputs; i < m_neurons.size(); i++)
740  // to
741  for (unsigned int j = 0; j < m_neurons.size(); j++) // from
742  {
743  int t_idx = ConnectionExists(i, j);
744  if (t_idx != -1)
745  {
746  //double t_derivative = unsigned_sigmoid_derivative( m_neurons[k].m_activation );
747  double t_derivative = 0;
748  if (m_neurons[k].m_activation_function_type
749  == NEAT::UNSIGNED_SIGMOID)
750  {
751  t_derivative = unsigned_sigmoid_derivative(
752  m_neurons[k].m_activation);
753  }
754  else if (m_neurons[k].m_activation_function_type
755  == NEAT::TANH)
756  {
757  t_derivative = tanh_derivative(
758  m_neurons[k].m_activation);
759  }
760 
761  double t_sum = 0;
762  // calculate the other sum
763  for (unsigned int l = 0; l < m_neurons.size(); l++)
764  {
765  int t_l_idx = ConnectionExists(k, l);
766  if (t_l_idx != -1)
767  {
768  t_sum += m_connections[t_l_idx].m_weight
769  * m_neurons[l].m_sensitivity_matrix[i][j];
770  }
771  }
772 
773  if (i == k)
774  {
775  t_sum += m_neurons[j].m_activation;
776  }
777  m_neurons[k].m_sensitivity_matrix[i][j] = t_derivative
778  * t_sum;
779  }
780  else
781  {
782  m_neurons[k].m_sensitivity_matrix[i][j] = 0;
783  }
784  }
785 
786  }

90  // Returns a random number from a gaussian (normal) distribution in the range of [-1 .. 1]
91  double RNG::RandGaussSigned()
92  {
93  static int t_iset = 0;
94  static double t_gset;
95  double t_fac, t_rsq, t_v1, t_v2;
96 
97  if (t_iset == 0)
98  {
99  do
100  {
101  t_v1 = 2.0 * (RandFloat()) - 1.0;
102  t_v2 = 2.0 * (RandFloat()) - 1.0;
103  t_rsq = t_v1 * t_v1 + t_v2 * t_v2;
104  } while (t_rsq >= 1.0 || t_rsq == 0.0);
105 
106  t_fac = sqrt(-2.0 * log(t_rsq) / t_rsq);
107  t_gset = t_v1 * t_fac;
108  t_iset = 1;
109 
110  double t_tmp = t_v2 * t_fac;
111 
112  Clamp(t_tmp, -1.0, 1.0);
113  return t_tmp;
114  }
115  else
116  {
117  t_iset = 0;
118  double t_tmp = t_gset;
119  Clamp(t_tmp, -1.0, 1.0);
120  return t_tmp;
121  }
122  }

124  int RNG::Roulette(std::vector<double> &a_probs)
125  {
126  double t_marble = 0, t_spin = 0, t_total_score = 0;

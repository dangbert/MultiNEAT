49 Population::Population(const Genome& a_Seed, const Parameters& a_Parameters,
50  bool a_RandomizeWeights, double a_RandomizationRange, int a_RNG_seed)
51 {
52  m_RNG.Seed(a_RNG_seed);
53  m_BestFitnessEver = -std::numeric_limits<double>::infinity();
54  m_Parameters = a_Parameters;
55 
56  m_Generation = 0;
57  m_NumEvaluations = 0;
58  m_NextGenomeID = m_Parameters.PopulationSize;
59  m_NextSpeciesID = 1;
60  m_GensSinceBestFitnessLastChanged = 0;
61  m_GensSinceMPCLastChanged = 0;
62 
63  // Spawn the population
64  for(unsigned int i=0; i<m_Parameters.PopulationSize; i++)
65  {
66  Genome t_clone = a_Seed;
67  t_clone.SetID(i);
68  m_Genomes.push_back( t_clone );
69  }
70 
71  // Now now initialize each genome's weights
72  for(unsigned int i=0; i<m_Genomes.size(); i++)
73  {
74  if (a_RandomizeWeights)
75  {
76  bool is_invalid = true;
77  while (is_invalid)
78  {
79  m_Genomes[i].Randomize_LinkWeights(a_RandomizationRange, m_RNG);
80  // randomize the traits as well
81  m_Genomes[i].Randomize_Traits(a_Parameters, m_RNG);
82  // and mutate nodes one initial time
83  m_Genomes[i].Mutate_NeuronActivations_A(a_Parameters, m_RNG);
84  m_Genomes[i].Mutate_NeuronActivations_B(a_Parameters, m_RNG);
85  m_Genomes[i].Mutate_NeuronActivation_Type(a_Parameters, m_RNG);
86  m_Genomes[i].Mutate_NeuronTimeConstants(a_Parameters, m_RNG);
87  m_Genomes[i].Mutate_NeuronBiases(a_Parameters, m_RNG);
88 
89  // check in the population if there is a clone of that genome
90  is_invalid = false;
91  if (!m_Parameters.AllowClones)
92  {
93  for(unsigned int j=0; j<m_Genomes.size(); j++)
94  {
95  if (i != j) // don't compare the same genome
96  {
97  if (m_Genomes[i].CompatibilityDistance(m_Genomes[j], m_Parameters) < 0.000001) // equal genomes?
98  {
99  is_invalid = true;
100  break;
101  }
102  }
103  }
104  }
105 
106  // Also don't let any genome to fail the constraints
107  if (!is_invalid) // doesn't make sense to do the test if already failed
108  {
109  if (m_Genomes[i].FailsConstraints(a_Parameters))
110  {
111  is_invalid = true;
112  }
113  }
114  }
115  }
116 
117  m_Genomes[i].SetNeuronRecursionLimit(a_Parameters.NeuronRecursionLimit);
118  //m_Genomes[i].CalculateDepth();
119  }
120  // Speciate
121  Speciate();
122 
123  // set these phased search variables now since used in MutateGenome
124  if (m_Parameters.PhasedSearching)
125  {
126  m_SearchMode = COMPLEXIFYING;
127  }
128  else
129  {
130  m_SearchMode = BLENDED;
131  }
132 
133  // Initialize the innovation database
134  m_InnovationDatabase.Init(a_Seed);
135 
136  m_BestGenome = m_Species[0].GetLeader();
137 
138  Sort();
139 
140  // Set up the rest of the phased search variables
141  CalculateMPC();
142  m_BaseMPC = m_CurrentMPC;
143  m_OldMPC = m_BaseMPC;
144 
145  m_InnovationDatabase.m_Innovations.reserve(50000);
146 }

248 // Separates the population into species
249 // also adjusts the compatibility treshold if this feature is enabled
250 void Population::Speciate()
251 {
252  // iterate through the genome list and speciate
253  // at least 1 genome must be present
254  ASSERT(m_Genomes.size() > 0);
255 
256  // first clear out the species
257  m_Species.clear();
258 
259 
260  bool t_added = false;
261 
262  // NOTE: we are comparing the new generation's genomes to the representatives from the previous generation!
263  // Any new species that is created is assigned a representative from the new generation.
264  for(unsigned int i=0; i<m_Genomes.size(); i++)
265  {
266  t_added = false;
267 
268  // iterate through each species and check if compatible. If compatible, then add to the species.
269  // if not compatible, create a new species.
270  for(unsigned int j=0; j<m_Species.size(); j++)
271  {
272  Genome tmp = m_Species[j].GetRepresentative();
273  if (m_Genomes[i].IsCompatibleWith( tmp, m_Parameters ))
274  {
275  // Compatible, add to species
276  m_Species[j].AddIndividual( m_Genomes[i] );
277  t_added = true;
278 
279  break;
280  }
281  }
282 
283  if (!t_added)
284  {
285  // didn't find compatible species, create new species
286  m_Species.push_back( Species(m_Genomes[i], m_NextSpeciesID));
287  m_NextSpeciesID++;
288  }
289  }
290 
291  // Remove all empty species (cleanup routine for every case..)
292  ClearEmptySpecies();
293 }

311 // Calculates how many offspring each genome should have
312 void Population::CountOffspring()
313 {
314  ASSERT(m_Genomes.size() > 0);
315  ASSERT(m_Genomes.size() == m_Parameters.PopulationSize);
316 
317  double t_total_adjusted_fitness = 0;
318  double t_average_adjusted_fitness = 0;
319  Genome t_t;
320 
321  // get the total adjusted fitness for all individuals
322  for(unsigned int i=0; i<m_Species.size(); i++)
323  {
324  for(unsigned int j=0; j<m_Species[i].m_Individuals.size(); j++)
325  {
326  t_total_adjusted_fitness += m_Species[i].m_Individuals[j].GetAdjFitness();
327  }
328  }
329 
330  // must be above 0
331  ASSERT(t_total_adjusted_fitness > 0);
332 
333  t_average_adjusted_fitness = t_total_adjusted_fitness / static_cast<double>(m_Parameters.PopulationSize);
334 
335  // Calculate how much offspring each individual should have
336  for(unsigned int i=0; i<m_Species.size(); i++)
337  {
338  for(unsigned int j=0; j<m_Species[i].m_Individuals.size(); j++)
339  {
340  m_Species[i].m_Individuals[j].SetOffspringAmount( m_Species[i].m_Individuals[j].GetAdjFitness() / t_average_adjusted_fitness);
341  }
342  }
343 
344  // Now count how many offspring each species should have
345  for(unsigned int i=0; i<m_Species.size(); i++)
346  {
347  m_Species[i].CountOffspring();
348  }

374 // Updates the species
375 void Population::UpdateSpecies()
376 {
377  // search for the current best species ID if not at generation #0
378  int t_oldbestid = -1, t_newbestid = -1;
379  int t_oldbestidx = -1;
380  if (m_Generation > 0)
381  {
382  for(unsigned int i=0; i<m_Species.size(); i++)
383  {
384  if (m_Species[i].IsBestSpecies())
385  {
386  t_oldbestid = m_Species[i].ID();
387  t_oldbestidx = i;
388  }
389  }
390  ASSERT(t_oldbestid != -1);
391  ASSERT(t_oldbestidx != -1);
392  }
393 
394  for(unsigned int i=0; i<m_Species.size(); i++)
395  {
396  m_Species[i].SetBestSpecies(false);
397  }
398 
399  bool t_marked = false; // new best species marked?
400 
401  for(unsigned int i=0; i<m_Species.size(); i++)
402  {
403  // Reset the species and update its age
404  m_Species[i].IncreaseAgeGens();
405  m_Species[i].IncreaseGensNoImprovement();
406  m_Species[i].SetOffspringRqd(0);
407 
408  // Mark the best species so it is guaranteed to survive
409  // Only one species will be marked - in case several species
410  // have equally best fitness
411  if ((m_Species[i].GetBestFitness() >= m_BestFitnessEver) && (!t_marked))
412  {
413  m_Species[i].SetBestSpecies(true);
414  t_marked = true;
415  t_newbestid = m_Species[i].ID();
416  }
417  }
418 
419  // This prevents the previous best species from sudden death
420  // If the best species happened to be another one, reset the old
421  // species age so it still will have a chance of survival and improvement
422  // if it grows old and stagnates again, it is no longer the best one
423  // so it will die off anyway.
424  if ((t_oldbestid != t_newbestid) && (t_oldbestid != -1))
425  {
426  m_Species[t_oldbestidx].ResetAgeGens();
427  }
428 }

432 // the epoch method - the heart of the GA
433 void Population::Epoch()
434 {
435  // So, all genomes are evaluated..
436  for(Species & species : m_Species)
437  for(Genome & individual: species.m_Individuals)
438  individual.SetEvaluated();
439 
440  // Sort each species's members by fitness and the species by fitness
441  Sort();
442 
443  // Update species stagnation info & stuff
444  UpdateSpecies();
445 
447  // Preparation
449 
450  // Adjust the species's fitness
451  AdjustFitness();
452 
453  // Count the offspring of each individual and species
454  CountOffspring();
455 
456  // Incrementing the global stagnation counter, we can check later for global stagnation
457  m_GensSinceBestFitnessLastChanged++;
458  // Find and save the best genome and fitness
459  for(auto & species : m_Species)
460  {
461  // Update best genome info
462  species.m_BestGenome = species.GetLeader();
463 
464  for(Genome & individual: species.m_Individuals)
465  {
466  // Make sure all are evaluated as we don't run in realtime
467  individual.SetEvaluated();
468 
469  const double t_Fitness = individual.GetFitness();
470  if (m_BestFitnessEver < t_Fitness)
471  {
472  // Reset the stagnation counter only if the fitness jump is greater or equal to the delta.
473  if (fabs(t_Fitness - m_BestFitnessEver) >= m_Parameters.StagnationDelta)
474  {
475  m_GensSinceBestFitnessLastChanged = 0;
476  }
477 
478  m_BestFitnessEver = t_Fitness;
479  m_BestGenomeEver = individual;
480  }
481  }
482  }
483 
484  // Find and save the current best genome
485  double t_bestf = std::numeric_limits<double>::min();
486  for(auto & species : m_Species)
487  {
488  for(Genome & individual: species.m_Individuals)
489  {
490  if (individual.GetFitness() > t_bestf)
491  {
492  t_bestf = individual.GetFitness();
493  m_BestGenome = individual;
494  }
495  }
496  }
497 
498  // adjust the compatibility threshold
499  if (m_Parameters.DynamicCompatibility)
500  {
501  if ((m_Generation % m_Parameters.CompatTreshChangeInterval_Generations) == 0)
502  {
503  if (m_Species.size() > m_Parameters.MaxSpecies)
504  {
505  m_Parameters.CompatTreshold += m_Parameters.CompatTresholdModifier;
506  }
507  else if (m_Species.size() < m_Parameters.MinSpecies)
508  {
509  m_Parameters.CompatTreshold -= m_Parameters.CompatTresholdModifier;
510  }
511  }
512 
513  if (m_Parameters.CompatTreshold < m_Parameters.MinCompatTreshold)
514  m_Parameters.CompatTreshold = m_Parameters.MinCompatTreshold;
515  }
516

527  // A special case for global stagnation.
528  // Delta coding - if there is a global stagnation
529  // for dropoff age + 10 generations, focus the search on the top 2 species,
530  // in case there are more than 2, of course
531  if (m_Parameters.DeltaCoding)
532  {
533  if (m_GensSinceBestFitnessLastChanged > (m_Parameters.SpeciesMaxStagnation + 10))
534  {
535  // make the top 2 reproduce by 50% individuals
536  // and the rest - no offspring
537  if (m_Species.size() > 2)
538  {
539  // The first two will reproduce
540  m_Species[0].SetOffspringRqd( m_Parameters.PopulationSize/2 );
541  m_Species[1].SetOffspringRqd( m_Parameters.PopulationSize/2 );
542

544  for (unsigned int i=2; i<m_Species.size(); i++)
545  {
546  m_Species[i].SetOffspringRqd( 0 );
547  }
548 
549  // Now reset the stagnation counter and species age
550  m_Species[0].ResetAgeGens();
551  m_Species[1].ResetAgeGens();
552  m_GensSinceBestFitnessLastChanged = 0;
553  }
554  }

565  // Phased searching core logic
567  // Update the current MPC
568  CalculateMPC();
569  if (m_Parameters.PhasedSearching)
570  {
571  // Keep track of complexity when in simplifying phase
572  if (m_SearchMode == SIMPLIFYING)
573  {
574  // The MPC has lowered?
575  if (m_CurrentMPC < m_OldMPC)
576  {
577  // reset that
578  m_GensSinceMPCLastChanged = 0;
579  m_OldMPC = m_CurrentMPC;
580  }
581  else
582  {
583  m_GensSinceMPCLastChanged++;
584  }
585  }
586 
587 
588  // At complexifying phase?
589  if (m_SearchMode == COMPLEXIFYING)
590  {
591  // Need to begin simplification?
592  if (m_CurrentMPC > (m_BaseMPC + m_Parameters.SimplifyingPhaseMPCTreshold))
593  {
594  // Do this only if the whole population is stagnating
595  if (m_GensSinceBestFitnessLastChanged > m_Parameters.SimplifyingPhaseStagnationTreshold)
596  {
597  // Change the current search mode
598  m_SearchMode = SIMPLIFYING;
599 
600  // Reset variables for simplifying mode
601  m_GensSinceMPCLastChanged = 0;
602  m_OldMPC = std::numeric_limits<double>::max(); // Really big one
603 
604  // reset the age of species
605  for(unsigned int i=0; i<m_Species.size(); i++)
606  {
607  m_Species[i].ResetAgeGens();
608  }
609  }
610  }
611  }
612  else if (m_SearchMode == SIMPLIFYING)
613  // At simplifying phase?
614  {
615  // The MPC reached its floor level?
616  if (m_GensSinceMPCLastChanged > m_Parameters.ComplexityFloorGenerations)
617  {
618  // Re-enter complexifying phase
619  m_SearchMode = COMPLEXIFYING;
620 
621  // Set the base MPC with the current MPC
622  m_BaseMPC = m_CurrentMPC;
623 
624  // reset the age of species
625  for(unsigned int i=0; i<m_Species.size(); i++)
626  {
627  m_Species[i].ResetAgeGens();
628  }
629  }
630  }
631  }

680  {

683  }
684  }

688  // If the total amount of genomes reproduced is less than the population size,
689  // due to some floating point rounding error,
690  // we will add some bonus clones of the first species's leader to it
691 
692  unsigned int t_total_genomes = 0;

694  t_total_genomes += static_cast<unsigned int>(m_Specie.m_Individuals.size());
695 
696  if (t_total_genomes < m_Parameters.PopulationSize)
697  {
698  int t_nts = m_Parameters.PopulationSize - t_total_genomes;
699 
700  while(t_nts--)
701  {
702  ASSERT(m_Species.size() > 0);
703  Genome t_tg = m_Species[0].m_Individuals[0];
704  m_Species[0].AddIndividual(t_tg);
705  }
706  }
707 
708

710  // Increase generation number
711  m_Generation++;
712 
713  // At this point we may also empty our innovation database
714  // This is the place where we control whether we want to
715  // keep innovation numbers forever or not.
716  if (!m_Parameters.InnovationsForever)
717  {
718  m_InnovationDatabase.Flush();
719  }

51 inline void GetMaxMin(const std::vector<double>& a_Vals, double& a_Min, double& a_Max)
52 {
53  a_Max = std::numeric_limits<double>::min();
54  a_Min = std::numeric_limits<double>::max();
55  for(std::vector<double>::const_iterator t_It = a_Vals.begin(); t_It != a_Vals.end(); ++t_It)
56  {
57  const double t_CurrentVal = (*t_It);
58  if (t_CurrentVal > a_Max) a_Max = t_CurrentVal;

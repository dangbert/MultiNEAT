603  for(unsigned int ix = 0; ix < ix_max; ix++){
604  std::string sum_permute = toBinary(ix, coord.size());
605  std::vector<double> child_coords;
606  unsigned int child_param_len = sum_permute.length();
607  child_coords.reserve(child_param_len);
608  for(unsigned int sign_ix = 0; sign_ix < child_param_len; sign_ix++)
609  {
610  if(sum_permute[sign_ix] == '0')
611  {
612  child_coords.push_back(coord[sign_ix] + width/2.0);
613  }
614  else
615  {
616  child_coords.push_back(coord[sign_ix] - width/2.0);
617  }
618  children.push_back(
619  std::make_shared<nTree>(child_coords, width/2.0, lvl+1)
620  );
621  }
622  }

625  std::string toBinary(unsigned int n, unsigned int min_len)
626  {
627  std::string r;

629  {
630  r=(n%2==0 ?"0":"1")+r; n/=2;
631 
632  }

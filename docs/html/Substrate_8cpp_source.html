232 int Substrate::GetMinCPPNInputs()
233 {
234  // determine the dimensionality across the entire substrate
235  int cppn_inputs = GetMaxDims() * 2; // twice, because we query 2 points at a time
236 
237  // the distance input
238  if (m_with_distance)
239  {
240  cppn_inputs += 1;
241  }
242 
243  return cppn_inputs + 1; // always count the bias
244 }

246 int Substrate::GetMinCPPNOutputs()

267 int Substrate::GetMaxDims()
268 {
269  unsigned int max_dims = 0;
270  for(unsigned int i=0; i<m_input_coords.size(); i++)
271  {
272  if (max_dims < m_input_coords[i].size())
273  {
274  max_dims = m_input_coords[i].size();
275  }
276  }
277  for(unsigned int i=0; i<m_hidden_coords.size(); i++)
278  {
279  if (max_dims < m_hidden_coords[i].size())
280  {
281  max_dims = m_hidden_coords[i].size();
282  }
283  }
284  for(unsigned int i=0; i<m_output_coords.size(); i++)
285  {
286  if (max_dims < m_output_coords[i].size())
287  {
288  max_dims = m_output_coords[i].size();
289  }
290  }
291  return max_dims;

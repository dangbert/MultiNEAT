268  // this method performs fitness sharing
269  // it also boosts the fitness of the young and penalizes old species
270  void Species::AdjustFitness(Parameters &a_Parameters)
271  {
272  ASSERT(m_Individuals.size() > 0);
273 
274  // iterate through the members
275  for (unsigned int i = 0; i < m_Individuals.size(); i++)
276  {
277  double t_fitness = m_Individuals[i].GetFitness();
278 
279  // the fitness must be positive
280  //DBG(t_fitness);
281  ASSERT(t_fitness >= 0.0);
282 
283  // this prevents the fitness to be below zero
284  if (t_fitness <= 0.0) t_fitness = 0.0000000001;
285 
286  // this prevents nan or infinity to be fitness
287  if (std::isnan(t_fitness)) t_fitness = 0.0000000001;
288  if (std::isinf(t_fitness)) t_fitness = 0.0000000001;
289 
290  // update the best fitness and stagnation counter
291  if (t_fitness > m_BestFitness)
292  {
293  m_BestFitness = t_fitness;
294  m_GensNoImprovement = 0;
295  }
296 
297  // boost the fitness up to some young age
298  if (m_AgeGenerations < a_Parameters.YoungAgeTreshold)
299  {
300  t_fitness *= a_Parameters.YoungAgeFitnessBoost;
301  }
302 
303  // penalty for old species
304  if (m_AgeGenerations > a_Parameters.OldAgeTreshold)
305  {
306  t_fitness *= a_Parameters.OldAgePenalty;
307  }
308 
309  // extreme penalty if this species is stagnating for too long time
310  // one exception if this is the best species found so far
311  if (m_GensNoImprovement > a_Parameters.SpeciesMaxStagnation)
312  {
313  // the best species is always allowed to live
314  if (!m_BestSpecies)
315  {
316  // when the fitness is lowered that much, the species will
317  // likely have 0 offspring and therefore will not survive
318  t_fitness *= 0.0000001;
319  }
320  }
321 
322  unsigned int ms = m_Individuals.size();
323  ASSERT(ms > 0);
324  if (ms == 0) ms = 1;
325 
326  // Compute the adjusted fitness for this member
327  m_Individuals[i].SetAdjFitness(t_fitness / (double)(ms));
328  }
329  }

345  // Reproduce mates & mutates the individuals of the species
346  // It may access the global species list in the population
347  // because some babies may turn out to belong in another species
348  // that have to be created.
349  // Also calls Birth() for every new baby
350  void Species::Reproduce(Population &a_Pop, Parameters &a_Parameters, RNG &a_RNG)
351  {
352  Genome t_baby; // temp genome for reproduction
353 
354  unsigned int t_offspring_count = Rounded(GetOffspringRqd());
355  unsigned int elite_offspring = Rounded(a_Parameters.EliteFraction * m_Individuals.size());
356  if (elite_offspring < 1) // can't be 0
357  {
358  elite_offspring = 1;
359  }
360  // ensure we have a champ
361  unsigned int elite_count = 0;
362  // no offspring?! yikes.. dead species!
363  if (t_offspring_count == 0)
364  {
365  // maybe do something else?
366  return;
367  }
368 
370  // Reproduction
371 
372  // Spawn t_offspring_count babies
373  //bool t_champ_chosen = false;
374  bool t_baby_exists_in_pop = false;
375  while (t_offspring_count--)
376  {
377  // Select the elite first..
378 
379  if (elite_count < elite_offspring)
380  {
381  t_baby = m_Individuals[elite_count];
382  elite_count++;
383  }
384  else
385  {
386  do // - while the baby already exists somewhere in the new population or turned invalid in some way
387  {
388  // this tells us if the baby is a result of mating
389  bool t_mated = false;
390 
391  // There must be individuals there..
392  ASSERT(NumIndividuals() > 0);
393 
394  // for a species of size 1 we can only mutate
395  // NOTE: but does it make sense since we know this is the champ?
396  if (NumIndividuals() == 1)
397  {
398  t_baby = GetIndividual(a_Parameters, a_RNG);
399  t_mated = false;
400  }
401  // else we can mate
402  else
403  {
404  Genome t_mom = GetIndividual(a_Parameters, a_RNG);
405 
406  // choose whether to mate at all
407  // Do not allow crossover when in simplifying phase
408  if ((a_RNG.RandFloat() < a_Parameters.CrossoverRate) && (a_Pop.GetSearchMode() != SIMPLIFYING))
409  {
410  // get the father
411  Genome t_dad;
412  bool t_interspecies = false;
413 
414  // There is a probability that the father may come from another species
415  if ((a_RNG.RandFloat() < a_Parameters.InterspeciesCrossoverRate) &&
416  (a_Pop.m_Species.size() > 1))
417  {
418  // Find different species (random one) // !!!!!!!!!!!!!!!!!
419  int t_diffspec = a_RNG.RandInt(0, static_cast<int>(a_Pop.m_Species.size() - 1));
420  t_dad = a_Pop.m_Species[t_diffspec].GetIndividual(a_Parameters, a_RNG);
421  t_interspecies = true;
422  }
423  else
424  {
425  // Mate within species
426  t_dad = GetIndividual(a_Parameters, a_RNG);
427 
428  // The other parent should be a different one
429  // number of tries to find different parent
430  int t_tries = 1024;
431  if (!a_Parameters.AllowClones)
432  {
433  while (((t_mom.GetID() == t_dad.GetID()) ||
434  (t_mom.CompatibilityDistance(t_dad, a_Parameters) < COMPAT_EQUALITY_DELTA)) &&
435  (t_tries--))
436  {
437  t_dad = GetIndividual(a_Parameters, a_RNG);
438  }
439  }
440  else
441  {
442  while (((t_mom.GetID() == t_dad.GetID())) && (t_tries--))
443  {
444  t_dad = GetIndividual(a_Parameters, a_RNG);
445  }
446  }
447  t_interspecies = false;
448  }
449 
450  // OK we have both mom and dad so mate them
451  // Choose randomly one of two types of crossover
452  if (a_RNG.RandFloat() < a_Parameters.MultipointCrossoverRate)
453  {
454  t_baby = t_mom.Mate(t_dad, false, t_interspecies, a_RNG, a_Parameters);
455  }
456  else
457  {
458  t_baby = t_mom.Mate(t_dad, true, t_interspecies, a_RNG, a_Parameters);
459  }
460 
461  t_mated = true;
462  }
463  // don't mate - reproduce the mother asexually
464  else
465  {
466  t_baby = t_mom;
467  t_mated = false;
468  }
469  }
470 
471  // Mutate the baby
472  if ((!t_mated) || (a_RNG.RandFloat() < a_Parameters.OverallMutationRate))
473  {
474  MutateGenome(t_baby_exists_in_pop, a_Pop, t_baby, a_Parameters, a_RNG);
475  }
476 
477  // Check if this baby is already present somewhere in the offspring
478  // we don't want that
479  t_baby_exists_in_pop = false;
480  // Unless of course, we want clones to exist
481  if (!a_Parameters.AllowClones)
482  {
483  for (const Species & tmpSpecies : a_Pop.m_TempSpecies)
484  {
485  for (const Genome & individual : tmpSpecies.m_Individuals)
486  {
487  if (
488  (t_baby.CompatibilityDistance(individual,
489  a_Parameters) < COMPAT_EQUALITY_DELTA) // identical genome?
490  )
491  {
492  t_baby_exists_in_pop = true;
493  break;
494  }
495  }
496  }
497  }
498 
499  // In case we want to enforce always new individuals
500  if (a_Parameters.ArchiveEnforcement)
501  {
502  for (unsigned int i = 0; i < a_Pop.m_GenomeArchive.size(); i++)
503  {
504  if (
505  (t_baby.CompatibilityDistance(a_Pop.m_GenomeArchive[i],
506  a_Parameters) < COMPAT_EQUALITY_DELTA) // identical genome?
507  )
508  {
509  t_baby_exists_in_pop = true;
510  break;
511  }
512  }
513  }
514  }
515  while (t_baby_exists_in_pop || (t_baby.FailsConstraints(a_Parameters))); // end do
516  }
517 
518  // We have a new offspring now
519  // give the offspring a new ID
520  t_baby.SetID(a_Pop.GetNextGenomeID());
521  t_baby.SetNeuronRecursionLimit(a_Parameters.NeuronRecursionLimit);
522  a_Pop.IncrementNextGenomeID();
523 
524  // sort the baby's genes
525  t_baby.SortGenes();
526 
527  // clear the baby's fitness
528  t_baby.SetFitness(0);
529  t_baby.SetAdjFitness(0);
530  t_baby.SetOffspringAmount(0);
531 
532  t_baby.ResetEvaluated();
533 
534  // Archive the baby if needed
535  if (a_Parameters.ArchiveEnforcement)
536  {
537  a_Pop.m_GenomeArchive.push_back(t_baby);
538  }
539 
541  // put the baby to its species //
543 
544  // before Reproduce() is invoked, it is assumed that a
545  // clone of the population exists with the name of m_TempSpecies
546  // we will store results there.
547  // after all reproduction completes, the original species will be replaced back
548 
549  bool t_found = false;
550  std::vector<Species>::iterator t_cur_species = a_Pop.m_TempSpecies.begin();
551 
552  // No species yet?
553  if (t_cur_species == a_Pop.m_TempSpecies.end())
554  {
555  // create the first species and place the baby there
556  a_Pop.m_TempSpecies.emplace_back(t_baby, a_Pop.GetNextSpeciesID());
557  a_Pop.IncrementNextSpeciesID();
558  }
559  else
560  {
561  // try to find a compatible species
562 
563  Genome t_to_compare = t_cur_species->GetRepresentative();
564 
565  t_found = false;
566  while ((t_cur_species != a_Pop.m_TempSpecies.end()) && (!t_found))
567  {
568  if (t_baby.IsCompatibleWith(t_to_compare, a_Parameters))
569  {
570  // found a compatible species
571  t_cur_species->AddIndividual(t_baby);
572  t_found = true; // the search is over
573  }
574  else
575  {
576  // keep searching for a matching species
577  t_cur_species++;
578  if (t_cur_species != a_Pop.m_TempSpecies.end())
579  {
580  t_to_compare = t_cur_species->GetRepresentative();
581  }
582  }
583  }
584 
585  // if couldn't find a match, make a new species
586  if (!t_found)
587  {
588  a_Pop.m_TempSpecies.emplace_back(t_baby, a_Pop.GetNextSpeciesID());
589  a_Pop.IncrementNextSpeciesID();
590  }
591  }
592  }

713  }

718  {
719  MutateGenome(t_baby_is_clone, a_Pop, t_baby, a_Parameters, a_RNG);
720  }
721

723  // we don't want that
724  t_baby_exists_in_pop = false;

726  if (!a_Parameters.AllowClones)
727  {

730  for (unsigned int j = 0; j < a_Pop.m_Species[i].m_Individuals.size(); j++)

736  {
737  t_baby_exists_in_pop = true;
738  break;
739  }
740  }
741  }
742  }
743 
744  // In case we want to enforce always new individuals
745  if (a_Parameters.ArchiveEnforcement)
746  {
747  for (unsigned int i = 0; i < a_Pop.m_GenomeArchive.size(); i++)
748  {
749  if (
750  (t_baby.CompatibilityDistance(a_Pop.m_GenomeArchive[i],
751  a_Parameters) < COMPAT_EQUALITY_DELTA) // identical genome?
752  )
753  {
754  t_baby_exists_in_pop = true;
755  break;
756  }
757  }
758  }
759  }
760  while (t_baby_exists_in_pop || t_baby.FailsConstraints(a_Parameters)); // end do
761 
762 
763  // We have a new offspring now
764  // give the offspring a new ID
765  t_baby.SetID(a_Pop.GetNextGenomeID());
766  a_Pop.IncrementNextGenomeID();
767 
768  // sort the baby's genes
769  t_baby.SortGenes();
770 
771  // clear the baby's fitness
772  t_baby.SetFitness(0);
773  t_baby.SetAdjFitness(0);
774  t_baby.SetOffspringAmount(0);
775 
776  t_baby.ResetEvaluated();
777 
778  // In case of archiving, add the new baby to the archive
779  if (a_Parameters.ArchiveEnforcement)
780  {
781  a_Pop.m_GenomeArchive.push_back(t_baby);
782  }
783 
784  return t_baby;
785  }

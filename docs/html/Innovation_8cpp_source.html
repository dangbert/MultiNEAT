87 void InnovationDatabase::Init(std::ifstream& a_DataFile)
88 {
89  m_Innovations.clear();
90  m_NextInnovationNum = 0;
91  m_NextNeuronID = 0;
92 
93  std::string t_str;
94 
95  // search for InnovationDatabaseStart
96  do
97  {
98  a_DataFile >> t_str;
99  }
100  while (t_str != "InnovationDatabaseStart");
101 
102  // Read the last innov numbers
103  a_DataFile >> t_str;
104  a_DataFile >> m_NextInnovationNum;
105  a_DataFile >> t_str;
106  a_DataFile >> m_NextNeuronID;
107 
108 
109  // Read the database until InnovationDatabaseEnd is encountered
110  do
111  {
112  a_DataFile >> t_str;
113 
114  if (t_str == "Innovation")
115  {
116  // Read in the innovation
117  int t_id, t_from, t_to, t_innovtype, t_neurontype, t_nid;
118 
119  a_DataFile >> t_id;
120  a_DataFile >> t_innovtype;
121  a_DataFile >> t_from;
122  a_DataFile >> t_to;
123  a_DataFile >> t_neurontype;
124  a_DataFile >> t_nid;
125 
126  m_Innovations.emplace_back( Innovation(t_id, static_cast<InnovationType>(t_innovtype), t_from, t_to, static_cast<NeuronType>(t_neurontype), t_nid) );
127  }
128 
129  }
130  while( t_str != "InnovationDatabaseEnd");
131 }

151 // Checks the database if the innovation has already occured
152 // Returns the innovation id if true or -1 if false
153 // If it is a NEW_LINK innovation, in & out specify the neuron IDs being connected
154 // If it is a NEW_NEURON innovation, in & out specify the connection that was split
155 int InnovationDatabase::CheckInnovation(int a_In, int a_Out, InnovationType a_Type) const
156 {
157  ASSERT((a_In > 0) && (a_Out > 0));
158  ASSERT((a_Type == NEW_NEURON) || (a_Type == NEW_LINK));
159 
160  // search the list for a match
161  for(unsigned int i=0; i < m_Innovations.size(); i++)
162  {
163  if ((m_Innovations[i].FromNeuronID() == a_In) && (m_Innovations[i].ToNeuronID() == a_Out) && (m_Innovations[i].InnovType() == a_Type))
164  {
165  // match found?
166  return m_Innovations[i].ID();
167  }
168  }
169 
170  // not found
171  return -1;
172 }
